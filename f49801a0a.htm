  133. public GetHomerDistanceBetweenPlayers(p1,p2)
  134. {
  135.     new Float:pX1,Float:pY1,Float:pZ1,Float:pX2,Float:pY2,Float:pZ2;
  136.     new Float:tmpdis;
  137.     GetPlayerPos(p1,pX1,pY1,pZ1);
  138.     GetPlayerPos(p2,pX2,pY2,pZ2);
  139.     tmpdis = floatsqroot(floatpower(floatabs(floatsub(pX2,pX1)),2)+floatpower(floatabs(floatsub(pY2,pY1)),2)+floatpower(floatabs(floatsub(pZ2,pZ1)),2));
  140.     return floatround(tmpdis);
